  \begin{split}
    D^a_i t^a_i &=
    f^a_i + I^a_e t^e_i - I{'}^m_i t^a_m + I^m_e (2 t^{ea}_{mi} - t^{ea}_{im}) + t^e_m (2 v^{ma}_{ei} - v^{am}_{ei}) \\
    & v^{mn}_{ei} (2 t^{ea}_{mn} - t^{ae}_{mn}) + v^{ma}_{ef}(2 t^{ef}_{mi} - t^{ef}_{im})
  \end{split}
  \label{t1}
\end{align}

\begin{align}

  \end{split}
  \label{t2}
\end{align}
